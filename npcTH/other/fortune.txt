L_displaycutin:
	.@arg = getarg(0);
	if (.@arg < 4) cutin "고스트링카드",4;		// Ghostring Card
	else if (.@arg < 7) cutin "드롭프스카드",4;	// Drops Card
	else if (.@arg < 10) cutin "맨티스카드",4;	// Mantis Card
	else if (.@arg < 13) cutin "무낙카드",4;		// Munak Card
	else if (.@arg < 16) cutin "산타포링카드",4;	// Santa Poring Card
	else if (.@arg < 19) cutin "솔져스켈톤카드",4;// Soldier Skeleton Card
	else if (.@arg < 22) cutin "스켈레톤카드",4;	// Skeleton Card
	else if (.@arg < 25) cutin "에드가카드",4;	// Eddga Card
	else if (.@arg < 28) cutin "엔젤링카드",4;	// Angeling Card
	else if (.@arg < 31) cutin "포링카드",4;		// Poring Card
	else cutin "포포링카드",4;// Poporing Card
	return;
}
