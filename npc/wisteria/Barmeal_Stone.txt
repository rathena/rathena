prontera,163,147,4	script	Barmeal Vendor	891,{
	.@item = 1000103;

	mes "[Barmeal Vendor]";
	mes "Good Afternon";
	mes "Do you want to";
	mes "buy the tickets?";
	mes "It will cost you 1000 zeny";
	mes "how much do you want?";
	next;
	input .@item;
	if (.@item == 0) {
		mes "[Barmeal Vendor]";
		mes "You have cancelled the trade.";
		mes "Have a good day.";
		close;
	}
	set .@total_cost,.@item * 1000;
		if (Zeny < .@Total_cost) {
			mes "[Barmeal Vendor]";
			mes "Oh, whoa~!";
			mes "You don't have enough";
			mes "zeny to exchange. You need to have";
			mes "at least " + .@total_cost + " zeny.";
			close;
		}
	set Zeny, Zeny-.@total_cost;
	getitem 1000103,.@item*1; //Barmeal Ticket	
	mes "[Barmeal Vendor]";
	mes "Here are your tickets.";
	mes "Have a good day.";
	close;
	}
