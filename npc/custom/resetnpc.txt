//===== rAthena Script =======================================
//= Reset NPC
//===== Description: =========================================
//= Resets skills, stats, or both.
//===== Additional Comments: =================================
//= 1.0 First Version
//= 1.1 Optimized for the greater good. [Kisuka]
//= 1.2 Cleaning [Euphy]
//= 1.3 All statuses removed upon skill reset. [Euphy]
//= 1.4 Compressed Script, Added limit use option [Stolao]
//=	Changed set -> setarray, Improved text with F_InsertPlural
//= 1.5 Added sc_end_class to reset related status changes [sader1992]
//============================================================
prontera,169,180,2	script	Reset	10010,{
	//	 		Skills,	Stats,	Both,	Limit
	setarray .@Reset,	500000,	500000,	900000,	0;
	mes "Reset Stats: "+ callfunc("F_InsertComma",.@Reset[1]) +"z";
	mes "Reset Skills: "+ callfunc("F_InsertComma",.@Reset[0]) +"z";
	mes "Reset ทั้งคู่: "+ callfunc("F_InsertComma",.@Reset[2]) +"z";
	next;
	set .@i,(select("^FF3355Reset Skills:Reset Stats:Reset ทั้งคู่^000000:ยกเลิก"));
	if(.@i > 3) close;
	if (Zeny < .@Reset[.@i-1]) {
		mes "Zeny ไม่เพียงพอ";
		close;
	}
	Zeny -= .@Reset[.@i-1];
	if(.@i&1){
		sc_end_class;
		ResetSkill;
	}
	if(.@i&2) ResetStatus;
	mes "เรียบร้อยแล้ว!";
	close;
	
OnInit:
	waitingroom "Reset",0;
	end;
}
