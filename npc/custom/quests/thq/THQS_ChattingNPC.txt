yuno_in01,32,178,3	script	Ahlma	94,{
	mes "[Ahlma]";
	if (Event_THQS == 0) goto N_Member;

	mes "How may I help you?";

	menu "Save",-,"Quit",N_Quit;
	mes "[Ahlma]";
	mes "Ok, saved. Thank you ~ See you soon~";

	close;

	mes "Well if you think you are safe, good on you.";
	close;
