	next;
	input .pass$;
	if(.pass$ == "sammeepay"){
		switch(select(
			" ~ เซ็ต VIP ให้ Player 1 วัน: ~ ใช้คำสั่ง GM สำหรับ Player:~ ใช้คำสั่ง GM @reloadscript:~ ใช้คำสั่ง GM @reloaditemdb: ~ ^777777ยกเลิก^000000"
		)) {
		case 1:
			mes "[ Service Player ]";
			mes "ว่าไงต้องเป็น VIP ใช่ไหม";
			next;
			mes "[ Service Player ]";
			mes "งั้นให้เป็นเพิ่มเป็น 1 วัน ละกันนะ";
			atcommand "@vip +1d " + strcharinfo(0);
			close;

			mes "[ Service Player ]";

			input .@cmd$;
			atcommand .@cmd$;
			close;

			atcommand "@reloadscript";

		case 4:
			atcommand "@reloaditemdb";

		case 5:
			mes "[ Service Player ]";
			mes "งั้นมาใหม่นะ";

		}
		end;
	}else{

	}
}
