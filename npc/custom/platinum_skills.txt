prontera,128,200,6	script	Habilidades de Platina	94,{
	mes "[Habilidades de Platina]";
	mes "Eu posso habilitar para você suas Habilidades Extras (Especiais).";
	mes "Você gostaria de recebe-las agora?";
	next;
	if (select("Sim, Por gentileza:Não") == 2) {
		mes "[Habilidades de Platina]";
		mes "Tenha um bom dia... >.>";
		close;
	}
	callfunc "F_GetPlatinumSkills";
	mes "[Habilidades de Platina]";
	mes "Aqui está, tenha um bom jogo !";
	close;
}
