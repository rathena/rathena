// Triggers NPCs
-	script	ach_treasure#touch	-1,{
	end;
OnTouch:
	sscanf( strnpcinfo(0), "#wp%d", .@num );
	if (.@num == 0)
		end;
	if (!achievementexists(.@num)) // Don't trigger if the player has already completed.
		cloakoffnpcself( ("#tr" + .@num) );
	end;
}
