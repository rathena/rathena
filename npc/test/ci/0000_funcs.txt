function	script	AssertTrue	{
	if (!getarg(0)) {
		errormes "AssertTrue failed for " + getarg(1) + ".";
		return false;
	}
	return true;
}

function	script	AssertEquals	{
	if (getarg(0) != getarg(1)) {
		errormes "AssertEquals failed for " + getarg(2) + ": expected " + getarg(0) + ", got " + getarg(1) + ".";
		return false;
	}
	return true;
}
